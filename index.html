# ── BioAuth gate helper ──────────────────────────────────────
bioauth() {
  local LABEL="${1:-Authenticate}"
  local RESULT
  RESULT=$(termux-fingerprint 2>/dev/null)
  if echo "$RESULT" | grep -q "AUTH_RESULT_SUCCESS"; then
    return 0
  else
    termux-toast -s "BioAuth failed: $LABEL"
    return 1
  fi
}
# ── GATE 1: Entry ────────────────────────────────────────────
echo ""
